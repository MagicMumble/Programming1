# check if a patient has more than 1 Y value
def sequence(series):
    return set(series)

def dummy(series):
    return reduce(lambda x, y: str(x) + str(y), series)

def isOneY(series):
    return len(sequence(series)) == 1

cutDf = df[['patient_id','y']]
cutDf = cutDf.groupby('patient_id').y.agg([sequence, dummy, isOneY]).reset_index()

def isOneYByPatient(patient_id, df):
    return df[df['patient_id'] == patient_id].isOneY

# check if patient V1.924 has the same y value in all rows

from bokeh.plotting import figure, show
from bokeh.layouts import gridplot
from bokeh.models import Range1d

def make_plot(title, y, pat_values, timeseq, color):
    p = figure(title=title, background_fill_color="#fafafa")
    p.y_range=Range1d(-1500, 1500)
    
    p.line(timeseq, pat_values, color=color) # CHANGE THIS LINE

    p.xaxis.axis_label = 'time'
    p.yaxis.axis_label = 'signal'
    p.grid.grid_line_color="white"
    return p


def make_plot_array(g, pa, y):
